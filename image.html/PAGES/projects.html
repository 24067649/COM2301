CODE 2-PALINDROME CHECK

Coding Language- Java
This code allows a user to input a word and the program checks if the word is spelt the same vice versa to confirm that it is a palindrome

package palindromecheck;
import java.util.Scanner;
public class Palindromecheck {
public static void main(String[] args) {
Scanner zee=new Scanner (System.in);
System.out.print("enter word ");
String text=zee.next();
String reversedtext = new StringBuilder(text).reverse().toString();
if (text.equals(reversedtext)) {
System.out.print("the word is a palindrome :" +text); }
else {
System.out.print("the word is not a palindrome :" );
}
}
}
